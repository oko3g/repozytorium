var N = 8; // Liczebnoœæ zbioru. Nie wstawiaj liczb
           // wiêkszych od 9, bo mo¿esz siê nie
           // doczekaæ rozwi¹zania

var d = Array(N);

// Funkcja sprawdzaj¹ca uporz¹dkowanie w zbiorze
//-----------------------------------------------------
function Posortowane()
{
  var i;

  for(i = 0; i < N - 1; i++) if(d[i] > d[i+1]) return false;
  return true;
}

// Procedura tasuj¹ca zbiór
//-------------------------
function Tasuj()
{
  var i,i1,i2,x;

  for(i = 1; i <= 3 * N; i++)
  {
    i1 = Math.floor(Math.random() * N); i2 = Math.floor(Math.random() * N);
    x = d[i1]; d[i1] = d[i2]; d[i2] = x;
  }
}

//******************************************************

function main()
{
  var i,t;

// Najpierw wype³niamy tablicê d[] liczbami pseudolosowymi

  for(i = 0; i < N; i++) d[i] = Math.floor(Math.random() * 10000);
  t = "Przed sortowaniem:<BR><BR>";
  for(i = 0; i < N; i++) t += d[i] + " ";
  t += "<BR><BR>";

// Sortujemy

  while(!Posortowane()) Tasuj();

// Wyœwietlamy wynik sortowania

  t += "Po sortowaniu<BR><BR>";
  for(i = 0; i < N; i++) t += d[i] + " ";
  document.getElementById("t_out").innerHTML = t;
}
